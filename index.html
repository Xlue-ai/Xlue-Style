CSS Design Tokens

將以下 :root 宣告貼入任何新平台即可繼承完整 Xlue 品牌配色

/* ── Xlue Design System — CSS Tokens v1.0 ── */
:root {
  /* Brand Colors */
  --color-primary:        #007D7D;   /* hsl(180 100% 25%) */
  --color-primary-hover:  #00A7A2;   /* hsl(178 100% 32%) */
  --color-primary-light:  #009C9C;   /* hsl(180 100% 31%) */
  --color-primary-dark:   #005A5A;   /* hsl(180 100% 18%) */
  --color-gradient-start: #00948A;   /* gradient from */
  --color-gradient-end:   #3A5A56;   /* gradient to   */

  /* Background Layers */
  --bg-l0:   #FAFAF8;   /* body — warm off-white    */
  --bg-l1:   #EFF4F4;   /* cards / sections         */
  --bg-l2:   #F4F7F6;   /* footer                   */
  --bg-card: #FFFFFF;   /* pure white card          */

  /* Text Colors */
  --text-primary:   #0D2B2B;          /* hsl(180 50% 10%) */
  --text-secondary: #2C4444;          /* hsl(180 30% 22%) */
  --text-tertiary:  #5C6C6C;          /* hsl(180 9% 40%)  */
  --text-weak:      rgba(14,46,46,.8); /* 80% opacity      */

  /* Borders */
  --border-default: #D8E2E0;
  --border-dark:    rgba(255,255,255,.08);

  /* Shadows */
  --shadow-card:  0 4px 12px rgba(0,0,0,.15);
  --shadow-light: 0 2px  8px rgba(0,0,0,.08);

  /* Border Radius */
  --radius-sm:   6px;
  --radius-md:   8px;
  --radius-lg:   12px;
  --radius-xl:   16px;
  --radius-full: 9999px;

  /* Typography */
  --font-sans: "Inter", ui-sans-serif, system-ui, -apple-system,
               BlinkMacSystemFont, "Segoe UI", Roboto, "Helvetica Neue",
               Arial, "Noto Sans", sans-serif;
  --font-mono: "JetBrains Mono", ui-monospace, SFMono-Regular,
               Menlo, monospace;
}
Tailwind Arbitrary Values Quick Reference
/* Colors */
bg-[#007D7D]               → brand primary
hover:bg-[#00A7A2]         → hover state
text-[#0D2B2B]             → primary text
text-[#2C4444]             → secondary text
text-[#5C6C6C]             → tertiary text
bg-[#EFF4F4]               → surface L1
bg-[#F4F7F6]               → footer
border-[#D8E2E0]           → default border
/* Patterns */
// Primary button
bg-[#007D7D] hover:bg-[#00A7A2] text-white

// Gradient text
bg-gradient-to-r from-[#00948A] to-[#3A5A56]
bg-clip-text text-transparent

// Card
bg-[#EFF4F4] rounded-2xl p-8

// Nav
